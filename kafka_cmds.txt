docker run -d -p 9092:9092 --name single-broker apache/kafka:latest

kafka-topics.sh --bootstrap-server localhost:9092 --list

kafka-topics.sh --bootstrap-server localhost:9092 --topic orders --create

kafka-topics.sh --bootstrap-server localhost:9092 --topic orders --describe

kafka-console-consumer.sh --bootstrap-server localhost:9092 --topic orders --from-beginning

kafka-console-producer.sh --bootstrap-server localhost:9092 --topic orders

docker-compose up -d

docker-compose logs -f

while true; do
  KEY="user-$((RANDOM % 6 + 1))"
  VALUE="event-$((RANDOM % 1000))"
  echo "$KEY:$VALUE" | kafka-console-producer.sh \
    --bootstrap-server localhost:29092 \
    --topic activity \
    --property "parse.key=true" \
    --property "key.separator=:"
  sleep 5
done
